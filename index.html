  <h1>Roulette Wheel</h1>

  <div id="names">

    <input type="text" id="nameInput" placeholder="Enter a name">

    <button onclick="addName()">Add Name</button>

  </div>

  <div id="wheel"></div>

  <button onclick="spinWheel()">Spin the Wheel</button>

  <div id="winner"></div>

 

  <script>

    let names = [];

    const wheel = document.getElementById('wheel');

    const winnerDiv = document.getElementById('winner');

 

    function addName() {

      const name = document.getElementById('nameInput').value.trim();

      if (name) {

        names.push(name);

        document.getElementById('nameInput').value = '';

        updateWheel();

      }

    }

 

    function updateWheel() {

      wheel.innerHTML = '';

      const sliceAngle = 360 / names.length;

      names.forEach((name, index) => {

        const slice = document.createElement('div');

        slice.className = 'slice';

        slice.style.backgroundColor = `hsl(${index * (360 / names.length)}, 70%, 50%)`;

        slice.style.transform = `rotate(${index * sliceAngle}deg)`;

        slice.textContent = name;

        wheel.appendChild(slice);

      });

    }

 

    function spinWheel() {

      if (names.length === 0) return;

      const randomSpin = Math.floor(Math.random() * 3600) + 1800;

      wheel.style.transform = `rotate(${randomSpin}deg)`;

      setTimeout(() => {

        const winnerIndex = Math.floor(Math.random() * names.length);

        winnerDiv.textContent = `Winner: ${names[winnerIndex]}`;

      }, 3000);

    }

  </script>
